<script>
  // ── NAVIGATION ──
  function showPage(id) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.getElementById('page-' + id).classList.add('active');
    document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
    document.querySelectorAll('.nav-item').forEach(n => {
      if (n.getAttribute('onclick') && n.getAttribute('onclick').includes(id)) {
        n.classList.add('active');
      }
    });
    if (id === 'financeiro') renderChart();
  }

  // ── PLANNING TABS ──
  function showSection(id) {
    ['canvas','swot','plano'].forEach(s => {
      document.getElementById('section-' + s).style.display = s === id ? 'block' : 'none';
    });
    document.querySelectorAll('#page-planejamento .tab-btn').forEach((btn, i) => {
      btn.classList.toggle('active', ['canvas','swot','plano'][i] === id);
    });
  }

  // ── TASKS ──
  function toggleTask(el) {
    const check = el.querySelector('.task-check');
    const text = el.querySelector('.task-text');
    const tag = el.querySelector('.task-tag');
    if (check.classList.contains('done')) {
      check.classList.remove('done');
      text.classList.remove('done');
      tag.textContent = '🔥 Pendente';
      tag.className = 'task-tag tag-peach';
    } else {
      check.classList.add('done');
      text.classList.add('done');
      tag.textContent = '✓ Feito';
      tag.className = 'task-tag tag-green';
    }
  }

  // ── CHART ──
  function renderChart() {
    const data = [
      { m:'Dez', r: 19, d: 12 },
      { m:'Jan', r: 21, d: 13 },
      { m:'Fev', r: 18, d: 11 },
      { m:'Mar', r: 24, d: 14 },
      { m:'Abr', r: 23, d: 13 },
      { m:'Mai', r: 28, d: 12 },
    ];
    const max = 32;
    const wrap = document.getElementById('bar-chart');
    wrap.innerHTML = data.map(d => `
      <div class="chart-bar-col">
        <div style="display:flex;gap:3px;align-items:flex-end;flex:1;width:100%">
          <div class="bar" style="height:${(d.r/max)*100}%;background:var(--sage-dark);flex:1"></div>
          <div class="bar" style="height:${(d.d/max)*100}%;background:var(--peach);flex:1"></div>
        </div>
        <div class="bar-label">${d.m}</div>
      </div>
    `).join('');
  }

  // ── MODAL ──
  function openModal() {
    document.getElementById('modal').classList.add('open');
  }
  function closeModal() {
    document.getElementById('modal').classList.remove('open');
  }
  function createProject() {
    const name = document.getElementById('proj-name').value || 'Novo Projeto';
    closeModal();
    showToast('Projeto "' + name + '" criado!');
  }

  // ── NOTIFICATIONS ──
  function toggleNotif() {
    document.getElementById('notif-panel').classList.toggle('open');
  }

  // ── TOAST ──
  let toastTimer;
  function showToast(msg) {
    const t = document.getElementById('toast');
    document.getElementById('toast-text').textContent = msg;
    t.classList.add('show');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(() => t.classList.remove('show'), 3000);
  }

  // ── AI CHAT ──
  const aiReplies = [
    "Ótima ideia! Com base nos dados do seu negócio, sugiro começar com uma análise de cohort dos seus clientes mais rentáveis. Isso pode revelar padrões de compra e ajudar a criar ofertas mais assertivas.",
    "Analisando seu histórico financeiro, identifico uma oportunidade de aumentar a margem em 8% reduzindo custos de infraestrutura e renegociando com fornecedores. Posso gerar um plano detalhado?",
    "Baseado em empresas similares à sua, o melhor momento para lançar uma campanha de captação é nos meses de março e setembro. Quer que eu crie um calendário editorial personalizado?",
    "Excelente pergunta! A validação de startup mais eficaz começa com entrevistas com 20 potenciais clientes. Posso te ajudar a criar um roteiro de perguntas e uma planilha de análise?",
    "Com R$ 28.540 de receita e 47 clientes, seu LTV médio está em R$ 607. Para aumentar para R$ 900, recomendo uma estratégia de expansão de conta — vou detalhar os próximos passos."
  ];
  let replyIdx = 0;

  function sendMsg() {
    const input = document.getElementById('chat-input');
    const msg = input.value.trim();
    if (!msg) return;
    addMsg(msg, 'user');
    input.value = '';
    setTimeout(() => addMsg(aiReplies[replyIdx % aiReplies.length], 'ai'), 900);
    replyIdx++;
  }

  function sendQuickMsg(msg) {
    showPage('ia');
    setTimeout(() => {
      addMsg(msg, 'user');
      setTimeout(() => addMsg(aiReplies[replyIdx % aiReplies.length], 'ai'), 900);
      replyIdx++;
    }, 50);
  }

  function addMsg(text, role) {
    const container = document.getElementById('chat-container');
    const div = document.createElement('div');
    div.className = 'chat-msg ' + role;
    div.innerHTML = `
      <div class="chat-av ${role}">${role === 'ai' ? '✦' : 'M'}</div>
      <div class="chat-bubble ${role}">${text}</div>
    `;
    container.appendChild(div);
    container.scrollTop = container.scrollHeight;
  }

  // ── INIT ──
  document.addEventListener('DOMContentLoaded', () => {
    renderChart();
  });
